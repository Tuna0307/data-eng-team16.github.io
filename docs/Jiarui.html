types <- levels(summary_data$flat_type)

total_by_year <- summary_data %>% group_by(year) %>% summarise(total = sum(transaction_count))

bar_traces <- list()
for (i in seq_along(types)) {
  type <- types[i]
  df <- summary_data %>% filter(flat_type == type)
  bar_traces[[i]] <- list(
    x = as.character(df$year),
    y = df$transaction_count,
    name = paste(type, "Volume"),
    type = 'bar',
    marker = list(opacity = 0.7),
    visible = TRUE
    # (market share text and hovertemplate removed)
  )
}

# Line traces for median price
line_traces <- lapply(seq_along(types), function(i) {
  type <- types[i]
  df <- summary_data %>% filter(flat_type == type)
  list(
    x = as.character(df$year),
    y = df$median_price,
    name = paste(type, "Median Price"),
    yaxis = 'y2',
    type = 'scatter',
    mode = 'lines+markers',
    line = list(width = 3),
    visible = TRUE
  )
})

all_traces <- c(bar_traces, line_traces)

fig <- plot_ly()
for (trace in all_traces) {
  fig <- fig %>% add_trace(
    x = trace$x,
    y = trace$y,
    name = trace$name,
    type = trace$type,
    marker = trace$marker,
    yaxis = trace$yaxis,
    mode = trace$mode,
    line = trace$line,
    visible = trace$visible
    # (text, textposition, textfont, hovertemplate all removed)
  )

# 1. volume bars (stacked)
vol_fig <- plot_ly() %>% 
  add_bars(data = summary_data, x = ~year, y = ~transaction_count,
           color = ~flat_type, legendgroup = ~flat_type,
           hovertemplate = "Year %{x}<br>%{y} transactions<extra></extra>") %>% 
  layout(barmode = "stack",
         yaxis = list(title = "Transactions"),
         showlegend = FALSE)

# 2. price lines
price_fig <- plot_ly() %>% 
  add_lines(data = summary_data, x = ~year, y = ~median_price,
            color = ~flat_type, legendgroup = ~flat_type,
            line = list(width = 2),
            hovertemplate = "Year %{x}<br>$%{y:,.0f}<extra></extra>") %>% 
  layout(yaxis = list(title = "Median Price", tickprefix = "$"))

# 3. combine
subplot(vol_fig, price_fig, nrows = 2, shareX = TRUE, heights = c(.58,.42)) %>%
  layout(title = "HDB Resale Volume & Price Evolution (2020-2024)",
         legend = list(orientation = 'h', y = -0.1)) %>% 
  config(displayModeBar = FALSE)
